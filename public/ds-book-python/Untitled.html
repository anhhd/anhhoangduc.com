    p = sns.boxplot("sex", "total_bill", hue = "smoker", data = tips)
    # Thêm title
    # Điều chỉnh tên trục

    sns.despine()
    p.legend(loc = 'upper center', ncol = 2, bbox_to_anchor = (0.5, 1.02))
    #plt.savefig("seaborn_chart.png")
